def self.parse (args)
  options = {}
  parser = OptionParser.new do |opt|
    opt.banner = "Usage: darkshadow pate [options]\nExample: darkshadow pate -l 50 -s ABC,def,123\nAd1Ad2Ad3Ae1Ae2Ae3Af1Af2Af3Bd1Bd2Bd3Be1Be2Be3Bf1Bf"
    opt.separator ''
    opt.separator 'Options:'
    opt.on('-l', '--length <length>', Integer, "The length of the pattern") do |len|
      options[:length] = len
    end

    opt.on('-s', '--sets <ABC,def,123>', Array, "Custom Pattern Sets") do |sets|
      options[:sets] = sets
    end

    opt.on_tail('-h', '--help', 'Show this message') do
      $stdout.puts opt
      exit
    end
  end

  parser.parse!(args)

  if options.empty?
    raise OptionParser::MissingArgument, 'No options set, try -h for usage'
  elsif options[:length].nil? && options[:sets]
    raise OptionParser::MissingArgument, '-l <length> is required'
  end

  options[:sets] = nil unless options[:sets]

  options
end
